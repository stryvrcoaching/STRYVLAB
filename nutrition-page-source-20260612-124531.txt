    25	function inferTrainingDay(protocolDay: Record<string, unknown> | null): boolean {
    26	  if (!protocolDay) return false
    27	  const name = String(protocolDay.name ?? '').toLowerCase()
    28	  const cycle = String(protocolDay.carb_cycle_type ?? '').toLowerCase()
    29	  return (
    30	    name.includes('entraînement') ||
    31	    name.includes('entrainement') ||
    32	    name.includes('training') ||
    33	    cycle.includes('high')
    34	  )
    35	}
