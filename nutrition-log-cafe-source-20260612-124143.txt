   492	    const h = new Date().getHours()

   500	    if (!drafts.length) return false

   535	  async function persistMeal(entriesDraft: EntryDraft[], mealSource: MealSource): Promise<boolean> {
   536	    if (!entriesDraft.length) return false
   537	    setSaving(true)
   538	    try {
   539	      const body: Record<string, unknown> = {
   540	        entries: entriesDraft.map(d => ({ food_item_id: d.food_item.id, quantity_g: d.quantity_g, input_mode: d.input_mode })),
   541	        meal_source: mealSource,
   542	      }
   543	      if (existingMealId) body.meal_id = existingMealId
   544	      // Pass logged_at as noon of the compose date so physiological_date is correct
   545	      // regardless of server timezone — mirrors the pattern used in savePrep()
   546	      if (prepDate) body.logged_at = `${prepDate}T12:00:00.000Z`
   547	      if (prepMealSlot) body.meal_type = prepMealSlot
   548	      const res = await fetch("/api/client/nutrition/meals", {
   549	        method: "POST",
   550	        headers: { "Content-Type": "application/json" },
   551	        body: JSON.stringify(body),
   552	      })
   553	      if (res.ok) {
   554	        if (onSuccess) onSuccess()
   555	        return true
   556	      }
   557	      setSaving(false)
   558	      return false
   559	    } catch {
   560	      setSaving(false)
   561	      return false
   562	    }
   563	  }
